// server.js (przykład minimalny) const express = require('express'); const app = express(); const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY); // ustaw w env app.use(express.json()); app.post('/create-checkout-session', async (req, res) => { const { price, email, product } = req.body; // przelicz na grosze const amount = Math.round(parseFloat(price) * 100); try { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], mode: 'payment', line_items: [{ price_data: { currency: 'eur', product_data: { name: product }, unit_amount: amount }, quantity: 1 }], customer_email: email, success_url: 'https://twojadomena.de/danke.html?session_id={CHECKOUT_SESSION_ID}', cancel_url: 'https://twojadomena.de/abbruch.html', }); res.json({ url: session.url }); } catch (err) { console.error(err); res.status(500).json({ error: 'Stripe error' }); } }); app.post('/fulfil-order', (req, res) => { // endpoint do obsługi powiadomień z klienta po PayPal (opcjonalny) console.log('Fulfil order', req.body); // wyślij e-mail z plikiem, zapisz w DB, itd. res.sendStatus(200); }); app.listen(4242, () => console.log('Server running on 4242'));
